// 1.刀耕火种时代，最经典的callback函数，多层嵌套即形成回调地狱
function doSomethingAfter(args, callback) {
    setTimeout( callback(args), 500 );
}
var result = doSomethingAfter('test', function(ret) {
    console.log(ret)
})

// node中最经典的场景
fs.readFile('/etc/passwd', 'utf-8', function (err, data) {
  if (err) throw err;
  console.log(data);

// 3.发布-订阅,这里就不手动实现了，可以参考vue的源码
export function eventsMixin (Vue: Class<Component>) {
  const hookRE = /^hook:/
  Vue.prototype.$on = function (event: string | Array<string>, fn: Function): Component {
    const vm = this
    if (Array.isArray(event)) {
      for (let i = 0, l = event.length; i < l; i++) {
        vm.$on(event[i], fn)
      }
    } else {
      (vm._events[event] || (vm._events[event] = [])).push(fn)
      if (hookRE.test(event)) {
        vm._hasHookEvent = true
      }
    }
    return vm
  }

  Vue.prototype.$emit = function (event: string): Component {
    const vm = this
    let cbs = vm._events[event]
    if (cbs) {
      cbs = cbs.length > 1 ? toArray(cbs) : cbs
      const args = toArray(arguments, 1)
      const info = `event handler for "${event}"`
      for (let i = 0, l = cbs.length; i < l; i++) {
        // 底层就是try{}catch(e){},可自行查阅
        invokeWithErrorHandling(cbs[i], vm, args, vm, info)
      }
    }
    return vm
  }
}

vm.$on('test', function (msg) {
  console.log(msg)
})

# async 函数是什么？一句话，它就是 Generator 函数的语法糖，关于Generator 函数大家可查阅阮一峰老师的《ECMAScript 6 入门》

// async await用法
async function timeout(ms) {
  await new Promise((resolve) => {
    setTimeout(resolve, ms);
  });
}

async function asyncPrint(value, ms) {
  await timeout(ms);
  console.log(value);
}

asyncPrint('hello world', 50);

/**
 * asyncFun
 * genFun: Generator 函数
 * 实现关注点：Generator + 自执行
 */
const asyncFun = (genFun) => {
    return new Promise((resolve, reject) => {
        const gen = genFun();
        const step = nextFun => {
            let next;
            try {
                next = nextFun();
            } catch(e) {
                return reject(e);
            }
            if(next.done) {
                return resolve(next.value);
            }
            Promise.resolve(next.value).then( val => {
                step(() => {return gen.next(val);})
            }, err => {
                step(() => {return gen.throw(err);})
            })
        }
        step(() => {return gen.next(undefined);})
    })
}
// eg:
asyncFun(function*(){
    let ret1 = yield new Promise(function(resolve,reject){
        setTimeout(() => {
            resolve('Hello');
        },1000);
    });
    console.log(ret1);
    var ret2 = yield new Promise(function(resolve,reject){
        setTimeout(function(){
            resolve('World');
        },1000);
    });
    console.log(ret2);
})
